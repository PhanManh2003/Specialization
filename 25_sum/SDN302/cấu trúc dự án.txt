my-app/
├── node_modules/
├── src/
│   ├── config/          # Cấu hình (DB, .env, logger, cors,...)
│   ├── controllers/     # Xử lý request/response, gọi service
│   ├── routes/          # Định nghĩa endpoint, gán controller
│   ├── models/          # Định nghĩa schema (Mongoose, Sequelize,...)
│   ├── services/        # Xử lý nghiệp vụ (business logic)
│   ├── repositories/    # Giao tiếp với DB (trừu tượng hóa truy vấn DB)
│   ├── database/        # Kết nối DB, migration, seeding (nếu có)
│   ├── middleware/      # Auth, validation, error handler,...
│   ├── common/          # Các helper, util, constant, formatter
│   └── app.js           # App chính
├── .env
├── .gitignore
├── package.json
└── server.js            # Khởi động server


route -> controller -> service -> repository -> database -> models
